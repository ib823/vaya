// vaya-pay/src/lib.rs // VayaPay Core - All regulatory compliance built-in pub struct VayaPayEngine { // Phase 1: Licensed PSP Integration payment_gateway: Box<dyn PaymentGateway>, // Phase 2: Credit/Discount System credit_ledger: CreditLedger, // Phase 3: Escrow Partner Integration escrow_client: EscrowClient, // Compliance (Always On) aml_engine: AmlEngine, audit_log: AuditLog, fraud_detector: FraudDetector, } impl VayaPayEngine { /// Process a booking payment pub async fn process_booking( &self, booking: &Booking, method: PaymentMethod, credits: Option<CreditAmount>, ) -> Result<PaymentResult> { // 1. AML/Fraud screening self.aml_engine.screen(&booking).await?; self.fraud_detector.check(&booking, &method).await?; // 2. Apply credits (discount, not payment) let net_amount = if let Some(c) = credits { self.credit_ledger.apply_discount(booking.user_id, c)?; booking.total - c.as_discount() } else { booking.total }; // 3. Process via licensed PSP let result = self.payment_gateway .charge(net_amount, &method) .await?; // 4. Audit log (compliance requirement) self.audit_log.record(&result).await?; Ok(result) } /// Process pool commitment via escrow pub async fn commit_to_pool( &self, pool: &DemandPool, user: &User, amount: Money, method: PaymentMethod, ) -> Result<CommitmentReceipt> { // Funds go to LICENSED TRUSTEE, not VAYA let escrow_ref = self.escrow_client .create_escrow( user.id, pool.id, amount, EscrowTerms { release_on: PoolActivated, refund_on: PoolFailed | PoolExpired, expiry: pool.expires_at, } ).await?; Ok(CommitmentReceipt { pool_id: pool.id, user_id: user.id, amount, escrow_reference: escrow_ref, trustee: "AmanahRaya Trustees Berhad", }) } /// Process instant refund (VAYA pre-funds) pub async fn instant_refund( &self, booking: &Booking, reason: RefundReason, ) -> Result<RefundResult> { // 1. Verify refund eligibility let amount = self.calculate_refund(booking, reason)?; // 2. VAYA pre-funds immediately let payout = self.payment_gateway .instant_payout(booking.user_id, amount) .await?; // 3. Create recovery claim from airline self.create_recovery_claim(booking, amount).await?; Ok(RefundResult { amount, status: RefundStatus::Completed, payout_id: payout.id, user_received_at: Timestamp::now(), }) } }

// vaya-pay/src/gateway.rs // Multi-PSP abstraction with automatic failover pub trait PaymentGateway: Send + Sync { async fn charge(&self, amount: Money, method: &PaymentMethod) -> Result<ChargeResult>; async fn refund(&self, charge_id: &str, amount: Money) -> Result<RefundResult>; async fn instant_payout(&self, user_id: UserId, amount: Money) -> Result<PayoutResult>; async fn tokenize_card(&self, card: CardDetails) -> Result<CardToken>; } /// iPay88 Implementation (Primary - Malaysian) pub struct IPay88Gateway { ... } impl PaymentGateway for IPay88Gateway { async fn charge(&self, amount: Money, method: &PaymentMethod) -> Result<ChargeResult> { match method { PaymentMethod::Fpx(bank) => self.charge_fpx(amount, bank).await, PaymentMethod::Card(token) => self.charge_card(amount, token).await, PaymentMethod::EWallet(wallet) => self.charge_ewallet(amount, wallet).await, PaymentMethod::Bnpl(provider) => self.charge_bnpl(amount, provider).await, } } } /// Stripe Implementation (Fallback - International) pub struct StripeGateway { ... } /// Multi-gateway with automatic failover pub struct MultiGateway { primary: Box<dyn PaymentGateway>, // iPay88 fallback: Box<dyn PaymentGateway>, // Stripe } impl PaymentGateway for MultiGateway { async fn charge(&self, amount: Money, method: &PaymentMethod) -> Result<ChargeResult> { match self.primary.charge(amount, method).await { Ok(result) => Ok(result), Err(e) => { // Log primary failure tracing::warn!("Primary PSP failed: {}, trying fallback", e); self.fallback.charge(amount, method).await } } } }
TECH-03 Credit Ledger (Double-Entry)
// vaya-pay/src/credits.rs // Double-entry bookkeeping for credit ledger pub struct CreditLedger { db: VayaStore, } impl CreditLedger { /// Award credits to user pub async fn award( &self, user_id: UserId, amount: CreditAmount, reason: AwardReason, ) -> Result<CreditTransaction> { let txn = CreditTransaction { id: TransactionId::new(), user_id, amount, txn_type: CreditTxnType::Award, reason: reason.to_string(), expires_at: Timestamp::now() + Duration::days(365), created_at: Timestamp::now(), }; // Double-entry: DEBIT expense, CREDIT user balance self.db.transaction(|tx| { tx.insert("credit_transactions", &txn)?; tx.increment("user_credit_balances", user_id, amount)?; tx.increment("credit_expense_account", "SYSTEM", amount)?; Ok(()) }).await?; Ok(txn) } /// Apply credits as discount (not payment) pub fn apply_discount( &self, user_id: UserId, amount: CreditAmount, ) -> Result<DiscountApplied> { // Check balance let balance = self.get_balance(user_id)?; if balance < amount { return Err(CreditError::InsufficientBalance); } // Check minimum (100 credits = RM 10) if amount.0 < 100 { return Err(CreditError::BelowMinimum); } // Deduct from balance self.db.transaction(|tx| { tx.decrement("user_credit_balances", user_id, amount)?; tx.decrement("credit_liability_account", "SYSTEM", amount)?; Ok(()) })?; Ok(DiscountApplied { credits_used: amount, discount_value: amount.as_discount(), // 10 credits = RM 1 }) } }
